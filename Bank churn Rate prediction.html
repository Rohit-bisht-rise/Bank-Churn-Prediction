def val_score(n):
    avg = []
    std = []
    for i in n:
        score = cross_val_score(KNN(n_neighbors=i), X=train_x, y=train_y, cv=10)
        avg.append(score.mean())
        std.append(score.std())
        print(score.mean())
    
    return avg, std
